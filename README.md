// To find the size of int,float,double and char
#include<iostream>
using namespace std;
int main(){
 	int a;
 	float b;
 	double c;
 	char d;
 	cout<<"Size of int :"<<sizeof(a)<<"\n";
 	cout<<"Size of float :"<<sizeof(b) <<"\n";
 	cout<<"Size of double :"<<sizeof(c)<<"\n";
 	cout<<"Size of char :"<<sizeof(d)<<"\n";
 	return 0;
}
